    <script>

        document.addEventListener("DOMContentLoaded", function() {

            // Plot dimensions

            const xCoord = 6;

            const yCoord = 6;

            

            // Container for heatmap plots

            const container = document.getElementById("container");

            

            // Sensor readings

            const moistureLevels = [3, 56, 4564, 7, 567, 658, 666, 6, 345, 345];

            

            // Generate heatmap

            for (let i = 0; i < xCoord * yCoord; i++) {

                const plot = document.createElement("div");

                plot.classList.add("plot");

                plot.dataset.x = (i % xCoord) + 1;

                plot.dataset.y = Math.floor(i / yCoord) + 1;

                

                // Determine moisture level and add corresponding class

                if (moistureLevels[i] < 650) {

                    plot.classList.add("dry");

                } else if (moistureLevels[i] <= 680) {

                    plot.classList.add("nice");

                } else {

                    plot.classList.add("wet");

                }

                

                container.appendChild(plot);

            }

        });

    </script>
